abstract public class C {
    private int a;
    int b;

    public int getA() {
        return a;
    }

    public void setA(int a) {
        this.a = a;
    }

    public int getB() {
        return b;
    }

    public void setB(int b) {
        this.b = b;
    }
    abstract public double laske(int x, int y);
}


//a
public class D extends C {
    //b
    private double d1;
    double d2;

    //c

    public void setD1(double d1) {
        this.d1 = d1;
    }

    public double getD1() {
        return d1;
    }

    public void setD2(double d2) {
        this.d2 = d2;
    }

    public double getD2() {
        return d2;
    }

    public double laske(int a, int b){
        return a*b;
    }


    public static void main(String[]args){
        D olio1 = new D();
        olio1.setD1(1);
        olio1.setD2(2);

        System.out.println(olio1.getD1());
        System.out.println(olio1.getD2());
        System.out.println(olio1.laske(3, 4));
    }
}

public class Kirjoitelma extends Tentti{
    int kielioppi;
    int oikeinkirjoitus;
    int pituus;
    int sisalto;

    public void setKielioppi(int kielioppi) {
        if (kielioppi <= 30 || kielioppi <0){
            this.kielioppi = kielioppi;
        } else {
            System.out.println("Tarkista Kieliopin pisteet.");
        }
    }
    // ii

    public void setOikeinkirjoitus(int oikeinkirjoitus) {
        if ( oikeinkirjoitus <= 20 || oikeinkirjoitus < 0) {
            this.oikeinkirjoitus = oikeinkirjoitus;
        } else {
            System.out.println("Tarkista oikeinkirjotuksen pisteet.");
        }
    }
    // iii

    public void setPituus(int pituus) {
        if (pituus <= 20 || pituus < 0) {
            this.pituus = pituus;
        } else {
            System.out.println("Tarkista pituuden pisteet.");
        }
    }

    // iv

    public void setSisalto(int sisalto){
        if (sisalto <= 30 || sisalto < 0) {
            this.sisalto = sisalto;
        } else {
            System.out.println("Tarkista sisallon pisteet");

        }
    }

    public int getKokonaispisteet(){
        return kielioppi + oikeinkirjoitus + pituus + sisalto;
    }

        Kirjoitelma kirjoitelma1 = new Kirjoitelma();
        kirjoitelma1.setKielioppi(20);
        kirjoitelma1.setOikeinkirjoitus(10);
        kirjoitelma1.setPituus(22);
        kirjoitelma1.setSisalto(27);
        System.out.println("Kokonaispisteet: " + kirjoitelma1.getKokonaispisteet());
        
    }
}

public interface Erinomainen {
    public boolean onkoErinomainen();
}



public class Tentti {
    private double pisteet;
    public void setPisteet(double p) {
        this.pisteet = p;
    }
    public double getPisteet(){
        return this.pisteet;
    }
    public boolean onkoErinomainen(){
        if (this.pisteet >= 90){
            return true;
        }
        else {
            return false;
        }
    }
    public int getArvosana(){
        int arvosana;

        if (this.pisteet >= 90) {
            arvosana = 5;
        }
        else if (this.pisteet >= 80){
            arvosana = 4;
        }
        else if (this.pisteet >= 70){
            arvosana = 3;
        }
        else if (this.pisteet >= 60){
            arvosana = 2;
        }
        else if (this.pisteet >= 50){
            arvosana = 1;
        }
        else {
            arvosana = 0;
        }
        return arvosana;
    }

    public static void main(String[]args){
        Tentti tentti1 = new Tentti();
        Tentti tentti2 =new Tentti();
        tentti1.setPisteet(77);
        tentti2.setPisteet(88);
        System.out.println("Pisteet: " + tentti1.onkoErinomainen());
        System.out.print("Pisteet: " + tentti2.onkoErinomainen());
    }
}
